  2. (a - b)2 = a2 - 2ab + b2

  3. a2 - b2 = (a + b) (a - b)

  4. a2 + b2 = (a + b)2 - 2ab

  5. a2 + b2 = (a - b)2 + 2ab

  6. a3 + b3 = (a + b)3 - 3ab(a + b)

  7. a3 - b3 = (a - b)3 + 3ab(a - b)
